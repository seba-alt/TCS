   ```python
   # For search events:

6. Applies pagination: `total = len(all_events)`, then slice `[offset:offset+limit]`.

export interface TimelineSearchEvent {

  type: 'click'
  expert_username: string

  search_query: string
  created_at: string  // ISO datetime
}

export interface LeadTimelineResponse {
  email: string
  events: TimelineEvent[]
  total: number
  limit: number
  offset: number
}
```
